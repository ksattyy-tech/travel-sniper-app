import streamlit as st
import requests

# هذا هو مفتاحك الذي أرسلته لي من RapidAPI
RAPID_API_KEY = "0a189d02cemsh7cf29bb265f5c0ep1917c5jsn8dcc2b79af10"
RAPID_API_HOST = "skyscanner44.p.rapidapi.com" 

st.set_page_config(page_title="Travel Sniper AI", page_icon="✈️", layout="wide")

st.title("✈️ صائد صفقات السفر الذكي")
st.write("نظام رصد الرحلات وحل مشكلات السفر آلياً")

# واجهة البحث
with st.container():
    col1, col2, col3 = st.columns(3)
    with col1:
        origin = st.text_input("من (رمز المطار):", "RUH")
    with col2:
        dest = st.text_input("إلى (رمز المطار):", "DXB")
    with col3:
        date = st.text_input("التاريخ (YYYY-MM-DD):", "2026-06-15")

if st.button("اقتناص أفضل سعر الآن 💰"):
    with st.spinner('جاري جلب البيانات من السيرفرات العالمية...'):
        url = f"https://{RAPID_API_HOST}/search-extended"
        querystring = {"origin": origin, "destination": destination if 'destination' in locals() else dest, "departureDate": date, "currency": "SAR"}
        headers = {"x-rapidapi-key": RAPID_API_KEY, "x-rapidapi-host": RAPID_API_HOST}
        
        try:
            response = requests.get(url, headers=headers, params=querystring)
            data = response.json()
            st.success("تم الاتصال بنجاح!")
            st.json(data) # لعرض البيانات الخام والتأكد من نجاح الربط
        except Exception as e:
            st.error(f"حدث خطأ: {e}")

st.divider()
st.caption("ملاحظة: تأكد من تفعيل اشتراك Skyscanner في RapidAPI (الخطة المجانية).")
